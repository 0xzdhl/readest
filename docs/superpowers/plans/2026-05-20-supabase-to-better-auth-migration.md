# Supabase Auth → Better-Auth Migration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace Supabase Auth (GoTrue) and PostgREST with better-auth + drizzle-orm directly against Postgres, keeping the same DB host. Six login methods, three client surfaces (web/Tauri/mobile), RLS preserved via native PG mechanisms.

**Spec:** `docs/superpowers/specs/2026-05-20-supabase-to-better-auth-migration-design.md` (read this first — every design decision is anchored there).

**Architecture:** Drizzle owns all schema; drizzle-kit emits a single `0000_init.sql` that replaces the existing 11 docker migrations. Server functions in TanStack Start authenticate via better-auth `getSession`, then open a transaction with `SET LOCAL app.user_id` so RLS policies enforce row ownership. Web uses better-auth cookie session; Tauri/mobile use the `bearer` plugin with a token stored in native secure storage. Object storage continues to use `@aws-sdk/client-s3` (S3 in prod, MinIO in dev) — only the auth check in front changes.

**Tech Stack:** TanStack Start `^1.167`, drizzle-orm + drizzle-kit, better-auth `^1.6.11`, `postgres-js` (Cloudflare-Workers-compatible Postgres driver), Resend SDK + nodemailer (dev → Mailpit), PostgreSQL with native RLS.

**Branch:** Working branch `refactor/migrate-to-better-auth` already exists (spec already committed there). All work lands as commits on this branch — single PR at the end.

## Cross-Cutting Conventions

These apply to **every** task in this plan. Do not skip.

1. **Run `pnpm dlx @tanstack/intent@latest list`** at the start of any task that creates or modifies TanStack Start routes / server functions. Then `pnpm dlx @tanstack/intent@latest load <package>#<skill>` if a match exists. Don't guess TanStack APIs from training data — they shift.
2. **Use ctx7 skill (`mcp__plugin_context7_context7__query-docs`)** before writing better-auth / drizzle-orm / drizzle-kit / Resend SDK code. Query the library by name, read the current docs.
3. **No `any` type.** Path alias is `@/*` → `./src/*`. Use existing patterns from sibling files.
4. **TDD where the unit makes it possible** (per `.claude/rules/test-first.md`): pure helpers, RLS policy assertions, server-function input/output. Schema definitions and config files can ship without a test if there's no behavior to test — but anything with a runtime branch gets a test first.
5. **Commit cadence:** at the end of each numbered Task (not each step). Commits use `CI=true git commit ... --no-verify` to skip the pnpm-install pre-commit hook (the user has authorized `--no-verify`). Pre-push hooks are similarly skipped.
6. **Done-conditions before committing:** run `pnpm lint` and `pnpm test -- --watch=false` in the `apps/readest-app/` dir; both must pass. Anything that touched `src-tauri/` also runs `pnpm fmt:check` + `pnpm clippy:check`.
7. **Working directory** for `pnpm` commands: `apps/readest-app/`. For root-level work (workspace deps, docker), use the repo root.

// apps/readest-app/src/db/rls.ts
import { sql } from 'drizzle-orm';
import { db } from './client';

type TxLike = Parameters<Parameters<typeof db.transaction>[0]>[0];

export async function withRls<T>(
  userId: string | null,
  fn: (tx: TxLike) => Promise<T>,
): Promise<T> {
  return db.transaction(async (tx) => {
    if (userId) {
      await tx.execute(sql`SELECT set_config('app.user_id', ${userId}, true)`);
    }
    return fn(tx);
  });
}

export async function withBypassRls<T>(
  fn: (tx: TxLike) => Promise<T>,
): Promise<T> {
  return db.transaction(async (tx) => {
    await tx.execute(sql`SELECT set_config('app.bypass_rls', 'true', true)`);
    return fn(tx);
  });
}
```

-- Connection role used by drizzle. Created idempotently.
DO $$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_roles WHERE rolname = 'readest_app') THEN
    CREATE ROLE readest_app LOGIN PASSWORD 'readest_app';
  END IF;
END
$$;
GRANT USAGE ON SCHEMA public TO readest_app;
GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA public TO readest_app;
GRANT USAGE, SELECT ON ALL SEQUENCES IN SCHEMA public TO readest_app;
ALTER DEFAULT PRIVILEGES IN SCHEMA public
  GRANT SELECT, INSERT, UPDATE, DELETE ON TABLES TO readest_app;
ALTER DEFAULT PRIVILEGES IN SCHEMA public
  GRANT USAGE, SELECT ON SEQUENCES TO readest_app;

-- Enable RLS on every business table and add the standard self-ownership policy.
-- (better-auth's user/session/account/verification tables are intentionally
--  left without RLS — the connection role manages them directly.)
DO $$
DECLARE t text;
BEGIN
  FOR t IN
    SELECT unnest(ARRAY[
      'books','book_configs','book_notes',
      'files',
      'book_shares',
      'replicas','replica_keys',
      'payments','subscriptions','customers',
      'apple_iap_subscriptions','google_iap_subscriptions'
    ])
  LOOP
    EXECUTE format('ALTER TABLE %I ENABLE ROW LEVEL SECURITY;', t);
    EXECUTE format($p$
      CREATE POLICY %I ON %I
        FOR ALL
        USING  (
          user_id = current_setting('app.user_id', true)::uuid
          OR current_setting('app.bypass_rls', true) = 'true'
        )
        WITH CHECK (
          user_id = current_setting('app.user_id', true)::uuid
          OR current_setting('app.bypass_rls', true) = 'true'
        );
    $p$, t || '_self', t);
  END LOOP;
END
$$;

import { Resend } from 'resend';
import nodemailer from 'nodemailer';

export interface SendEmailArgs {
  to: string;
  subject: string;
  html: string;
}

export async function sendEmail({ to, subject, html }: SendEmailArgs): Promise<void> {
  const apiKey = process.env.RESEND_API_KEY;
  const from = process.env.RESEND_FROM_EMAIL ?? 'noreply@readest.app';

  if (apiKey) {
    const resend = new Resend(apiKey);
    const { error } = await resend.emails.send({ from, to, subject, html });
    if (error) throw new Error(`Resend error: ${error.message}`);
    return;
  }

  // dev fallback: SMTP via Mailpit
  const host = process.env.SMTP_HOST ?? 'localhost';
  const port = Number(process.env.SMTP_PORT ?? 1025);
  const transport = nodemailer.createTransport({ host, port, secure: false });
  await transport.sendMail({ from, to, subject, html });
}

import { createServerFn } from '@tanstack/react-start';
import { getRequest } from '@tanstack/react-start/server';
import { auth, type Session } from '@/auth/server';
import { withRls, withBypassRls } from '@/db/rls';

export async function resolveSessionOr401(headers: Headers): Promise<NonNullable<Session>> {
  const session = await auth.api.getSession({ headers });
  if (!session) throw new Response('Unauthorized', { status: 401 });
  return session;
}

/** Protected: requires a session, opens an RLS transaction. */
export const protectedFn = createServerFn().middleware([
  async ({ next }) => {
    const request = getRequest();
    const session = await resolveSessionOr401(request.headers);
    return withRls(session.user.id, (tx) =>
      next({ context: { user: session.user, session, tx } }),
    );
  },
]);

/** Service: no session, RLS bypassed. For webhook signature-verified routes. */
export const serviceFn = createServerFn().middleware([
  async ({ next }) => withBypassRls((tx) => next({ context: { tx } })),
]);

/** Public: no session, no RLS context. */
export const publicFn = createServerFn();

import { createAuthClient } from 'better-auth/react';
import { magicLinkClient, bearerClient, inferAdditionalFields } from 'better-auth/client/plugins';
import type { auth } from './server';

const TOKEN_KEY = 'readest:bearer-token';

function loadToken(): string | null {
  try { return localStorage.getItem(TOKEN_KEY); } catch { return null; }
}
function storeToken(token: string | null): void {
  try {
    if (token) localStorage.setItem(TOKEN_KEY, token);
    else localStorage.removeItem(TOKEN_KEY);
  } catch { /* fail silently — Tauri WebView writes through to native storage */ }
}

export const nativeAuthClient = createAuthClient({
  baseURL: import.meta.env.VITE_BETTER_AUTH_URL ?? '',
  plugins: [
    magicLinkClient(),
    bearerClient(),
    inferAdditionalFields<typeof auth>(),
  ],
  fetchOptions: {
    auth: { type: 'Bearer', token: loadToken },
    onSuccess: (ctx) => {
      const setAuthHeader = ctx.response.headers.get('set-auth-token');
      if (setAuthHeader) storeToken(setAuthHeader);
    },
  },
});

export { storeToken, loadToken };
```

import { createContext, useContext, type ReactNode } from 'react';
import { authClient } from '@/auth/client';

interface AuthContextValue {
  user: ReturnType<typeof authClient.useSession>['data'] extends { user: infer U } ? U | null : null;
  isLoading: boolean;
  signOut: () => Promise<void>;
}

const AuthContext = createContext<AuthContextValue | null>(null);

export function AuthProvider({ children }: { children: ReactNode }) {
  const { data, isPending } = authClient.useSession();
  const value: AuthContextValue = {
    user: data?.user ?? null,
    isLoading: isPending,
    signOut: async () => { await authClient.signOut(); },
  };
  return <AuthContext.Provider value={value}>{children}</AuthContext.Provider>;
}

export function useAuth(): AuthContextValue {
  const ctx = useContext(AuthContext);
  if (!ctx) throw new Error('useAuth must be used within AuthProvider');
  return ctx;
}
